function processResponse(request) {
var books = JSON.parse(request.responseText);

for (var i = 0; i < books.length; i++) {
    var book = books[i];
    var p = document.createElement('p');
    p.innerHTML = book.title + ' by ' + book.author;
    document.body.appendChild(p);
  }
}
