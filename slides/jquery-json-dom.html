jQuery + JSON + DOM

Or we can avoid all of that JavaScript and use jQuery instead!

$(document).ready(function() {
  var aboutMe = {
    "firstName": "Lauren",
    "lastName": "Scott",
    "favoriteThings": ["haikus", "ruby"]
  };

  var $p = $('<p>');
  $p.html(aboutMe.firstName + ' ' + aboutMe.lastName);
  $p.append('My favorite things are ' + aboutMe.favorireThings.join(','));
});
